type int destination = 1,2,3,4
type int source = 4,3,2,1

type int readIndex = 0
while(readIndex < arrayLength(destination)) 
    printf(%s, "COPYING: ")
    printf(%i, source[readIndex])
    printf(%s "INTO: ")
    printf(%i, destination[readIndex])

    intcpy(destination[readIndex], source[readIndex])

    printf(%s, "RESULT AT ADDRESS: ")
    printf(%i, destination[readIndex])

    printNewLine()

    ++ readIndex
endWhile

printLine(%s, "--done--")
